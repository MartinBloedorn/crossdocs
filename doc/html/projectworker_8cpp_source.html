46  QDir dir(prjconffile);

63  return false;
64  }
65  project.tag = tempList[0];
66  project.name = tempList[0];

73  project.name = tempList[0];
74 
75  // Build engine -------------------------------------------
76  tempList = fp->getSectionContents(confsecBuildEngine);
77  if(tempList.isEmpty()) {
78  qWarning() << QString(__FUNCTION__) << "Undefined section "
79  << confsecBuildEngine << " Your documents should have it!";
80  project.buildEngine = CDC_buildEngine::none;
81  } else {
82  if(tempList[0].compare(QString("doxygen")))
83  project.buildEngine = CDC_buildEngine::doxygen;
84  else if(tempList[0].compare(QString("custom")))

91  qWarning() << QString(__FUNCTION__) << "Undefined section "

93  return false;
94  }

96  for(int i = 0; i < tempList.length(); ++i) {
97  documentWorker * pd = new documentWorker();

107  // Verify any repeated document tag

110  taglist.append(project.documents[i]->getTag());
111  for(int i = 0; i < project.documents.length(); ++i)
112  if(taglist.lastIndexOf(project.documents[i]->getTag()) != i) {
113  qWarning() << QString(__FUNCTION__) << "Repeated document tag: "
