171 void cdcMainWindow::messageHandler(QtMsgType type, const char *msg, bool isDialog)
172 {
173  if(isDialog)
174  {
175  switch(type) {
176  case QtDebugMsg: QMessageBox::warning(this, tr("Message"), QString(msg), QMessageBox::Ok); break;
177  case QtWarningMsg: QMessageBox::warning(this, tr("Warning"), QString(msg), QMessageBox::Ok); break;
178  case QtCriticalMsg:QMessageBox::warning(this, tr("Critical!"),QString(msg), QMessageBox::Ok); break;
179  case QtFatalMsg: QMessageBox::warning(this, tr("FATAL"), QString(msg), QMessageBox::Ok); break;
180  }
181  }
182  else
183  {
184  switch (type) {
185  case QtDebugMsg:
186  logArea->setTextColor(Qt::black);
187  logArea->append(QString(QDateTime::currentDateTime().toString("hh.mm.ss.zzz") + ": " + msg));
188  break;
189  case QtWarningMsg:
190  logArea->setTextColor(QColor(255,150,0));
191  logArea->append(QString(QDateTime::currentDateTime().toString("hh.mm.ss.zzz") + ": " + msg));
192  break;
193  case QtCriticalMsg:
194  logArea->setTextColor(Qt::red);
195  logArea->append(QString(QDateTime::currentDateTime().toString("hh.mm.ss.zzz") + ": " + msg));
196  break;
197  case QtFatalMsg:
198  logArea->setTextColor(Qt::red);
199  logArea->append(QString(QDateTime::currentDateTime().toString("hh.mm.ss.zzz") + " FATAL: " + msg));
200  break;
201  }
202  }
203  logArea->moveCursor(QTextCursor::End);
204 }
